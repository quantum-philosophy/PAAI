\begin{algorithm}
  \caption{\emph{Evaluate} an interpolant at a set of points.}
  \alab{evaluate}
  \begin{algorithmic}[1]
    \vspace{0.4em}

    \Require{surrogate} \Comment{Artifacts of interpolation}
    \Require{points} \Comment{Points of interest}
    \Ensure{estimates} \Comment{Estimated values}

    \vspace{0.4em}

    \Let{estimates}{New()}

    \For{point \textbf{in} points}
      \Let{estimate}{0}
      \For{(index, surplus) \textbf{in} surrogate}
        \Let{weight}{Interpolate(index, point)}
        \Let{estimate}{$\text{estimate} + \text{surplus} \times \text{weight}$}
      \EndFor
      \State Append(estimates, estimate)
    \EndFor

    \State \textbf{return} estimates

    \vspace{0.4em}
  \end{algorithmic}
\end{algorithm}
